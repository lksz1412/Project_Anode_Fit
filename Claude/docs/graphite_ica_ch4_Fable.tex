\section*{서론}
\addcontentsline{toc}{section}{서론}
% ====================================================================
흑연 음극에서 각 staging 전이가 \emph{진행하는 속도}는 그 전이의 계면에서 일어나는 전기화학 전하전달 반응이
정한다. 전이 진행률 $\xi_j$ 의 변화는 정방향(전이 완성, 탈리튬화)과 역방향(재리튬화) 두 반쪽 과정의 경쟁이며,
\emph{그 경쟁의 불균형}이 net 진행을 정하고 \emph{두 반쪽의 절대 크기}가 비가역 소산을 정한다. 같은 반응을
계면 전류로 보면 전류$\cdot$과전압$\cdot$교환전류$\cdot$전하전달 저항이라는 측정 가능한 양으로 드러난다.

본 장은 이 전기화학 반응속도론을 다음 순서로 전개한다. 먼저 정$\cdot$역 flux 와 그 공액 구동력(반응 친화도)을
세우고(\S\ref{sec:nch2_flux}), 이를 계면 전류로 옮겨 반응 과전압과 Butler--Volmer 전류식을 얻는다
(\S\ref{sec:nch2_bv}). 소신호 한계에서 전하전달 저항을, 평형 한계에서 교환전류밀도와 그 온도$\cdot$조성 의존을
유도한다(\S\ref{sec:nch2_rct}--\ref{sec:nch2_i0}). 끝으로 정상 진행률이 평형값에서 벗어나는 조건
(\S\ref{sec:nch2_ss})과 외부 전류의 전이별 분배(\S\ref{sec:nch2_current})를 본다. 전개는 방전(탈리튬화) 방향
$s=+1$ 을 기본으로 하며, 각 결과는 차원$\cdot$극한 검산과 평형$\cdot$가역 극한에서의 환원을 함께 보인다.

\section{정$\cdot$역 flux 와 flux--force 짝}\label{sec:nch2_flux}
% ====================================================================
전이 $j$ 의 진행률 변화 $\dd\xi_j/\dd t=r_j^+(1-\xi_j)-r_j^-\xi_j$(식~(1.20))의 우변 두 항은 각각 한 방향으로만
흐르는 \emph{단방향 flux}다. forward(전이 완성, 탈리튬화)는 아직 전이 안 된 분율 $(1-\xi_j)$ 에 정방향 속도
$r_j^+$ 가, backward(재리튬화)는 이미 전이된 분율 $\xi_j$ 에 역방향 속도 $r_j^-$ 가 작용한 것이다:
\begin{equation}
\mathcal J_j^+\equiv r_j^+(1-\xi_j),\qquad \mathcal J_j^-\equiv r_j^-\xi_j,\qquad
\frac{\dd\xi_j}{\dd t}=\mathcal J_j^+-\mathcal J_j^-.
\label{eq:nch2_Jpm}
\end{equation}

\textbf{왜 net 이 아니라 정$\cdot$역을 따로 보는가.} 진행률만 따지면 net flux $\mathcal J_j^+-\mathcal J_j^-$ 로
충분하다. 그러나 반응이 만드는 \emph{비가역성}은 net 이 아니라 \emph{두 단방향 flux 의 불균형}이 정한다.
평형에서는 $\mathcal J_j^+=\mathcal J_j^-$ 라 net 은 $0$ 이지만 두 flux 자체는 $0$ 이 아니며(정$\cdot$역이 같은
크기로 끊임없이 오간다), \emph{그 둘이 같아질 때만} 비가역 생성이 사라진다. 따라서 net 으로 합치기 전의 단방향
flux 를 살려 두어야 소산을 옳게 센다.

\textbf{공액 구동력의 유도.} 비평형 열역학에서 화학 반응의 entropy 생성은 \emph{flux} 와 \emph{공액 affinity} 의
곱으로 적힌다 \cite{degrootmazur1962,prigogine1967}. 그 affinity 는 두 단방향 flux 비의 로그에서 나온다. 정$\cdot$역
속도비는 \emph{detailed balance} 가 정하므로(식~(1.29), $r_j^+/r_j^-=e^{\mathcal A_j/RT}$), flux 비는
\begin{equation}
\frac{\mathcal J_j^+}{\mathcal J_j^-}=\frac{r_j^+}{r_j^-}\cdot\frac{1-\xi_j}{\xi_j}
=e^{\mathcal A_j/RT}\,\frac{1-\xi_j}{\xi_j}
\label{eq:nch2_Jratio}
\end{equation}
다(첫 등호는 정의~\eqref{eq:nch2_Jpm}, 둘째는 식~(1.29) 대입; 구동력 $\mathcal A_j=sF(V_\drive-U_j)$ 는
식~(1.27)). 내부 상태 $0<\xi_j<1$, $r_j^\pm>0$ 을 가정하며(끝점은 극한으로 처리), 양변에 $RT\ln(\cdot)$ 을
취하면 $\ln(ab)=\ln a+\ln b$ 로
\begin{equation}
\boxed{\;\mathcal A_j^{\mathrm{net}}\equiv RT\ln\frac{\mathcal J_j^+}{\mathcal J_j^-}
=\mathcal A_j-RT\ln\frac{\xi_j}{1-\xi_j}\;,}
\label{eq:nch2_Anet}
\end{equation}
즉 net 반응을 구동하는 유효 친화도 $\mathcal A_j^{\mathrm{net}}$ 는 \emph{중심 기준} 구동력 $\mathcal A_j$ 에서
현재 점유율의 configurational 항 $RT\ln[\xi_j/(1-\xi_j)]$ 을 뺀 것이다. 앞 항은 ``전이 중심에서 얼마나 벗어났나''
(열역학 구동), 뒤 항은 ``현재 얼마나 진행됐나''(엔트로피적 되돌림)를 재며, 둘의 차가 실제 net 구동이다.

\begin{verifybox}
\textbf{평형 검산.} 평형 $\xi_j=\xi_{\eq,j}$ 에서는 식~(1.11) logistic 의 logit 이
$\ln[\xi_{\eq,j}/(1-\xi_{\eq,j})]=\mathcal A_j/RT$(기본형 $V_\drive=V_n$, $w_j=RT/F$)이므로
$\mathcal A_j^{\mathrm{net}}=\mathcal A_j-\mathcal A_j=0$ — net 친화도가 $0$ 이라 정$\cdot$역 flux 가 같아지고
($\mathcal J_j^+=\mathcal J_j^-$) net 진행과 소산이 동시에 멎는다. \textbf{차원.} $\mathcal J_j^\pm$ 비는 무차원,
$RT\ln(\cdot)=\mathrm{J/mol}=\mathcal A_j$ 와 정합.
\end{verifybox}

\textbf{소산의 부호(De Donder).} net flux $(\mathcal J_j^+-\mathcal J_j^-)$ 와 그 공액력 $\mathcal A_j^{\mathrm{net}}
=RT\ln(\mathcal J_j^+/\mathcal J_j^-)$ 는 \emph{같은 부호}다($a>b\Leftrightarrow\ln(a/b)>0$). 따라서
\begin{equation}
(\mathcal J_j^+-\mathcal J_j^-)\,\mathcal A_j^{\mathrm{net}}\ \ge\ 0,
\label{eq:nch2_dedonder}
\end{equation}
등호는 평형($\mathcal J_j^+=\mathcal J_j^-$)에서만 성립한다. 이는 ``net 반응은 친화도 내리막으로만 일어난다''는
2법칙의 반응 형태(De Donder 부등식)이며 \cite{degrootmazur1962}, 정$\cdot$역 flux 를 살려 둔 덕에 부호가 항별로
닫힌다.

\section{계면 실현 — 반응 과전압과 Butler--Volmer 전류}\label{sec:nch2_bv}
% ====================================================================
식~\eqref{eq:nch2_Anet} 의 $\mathcal A_j^{\mathrm{net}}$ 는 \emph{현재 상태} $\xi_j$ 가 \emph{국소 평형}에서 벗어난
정도를 몰당 에너지로 잰 것이다. 전기화학에서 같은 양을 \emph{전위} 단위로 적은 것이 반응 과전압 $\eta_j$ 다.

\textbf{국소 평형 전위.} 전이 $j$ 가 현재 진행률 $\xi_j$ 에서 평형이 되려면, 평형 진행률 식~(1.11)이 바로 그 $\xi_j$
를 주는 전위여야 한다. $\xi_j=[1+e^{-s(V-U_j)/w_j}]^{-1}$ 을 $V$ 에 대해 풀면($e^{-s(V-U_j)/w_j}=(1-\xi_j)/\xi_j$
에서 로그)
\begin{equation}
V_{\eq,j}(\xi_j)=U_j+\frac{w_j}{s}\,\ln\frac{\xi_j}{1-\xi_j}
\label{eq:nch2_Veq}
\end{equation}
이 \emph{국소 평형 전위}다 — 전이 중심 $U_j$ 에서 현재 점유율만큼 logit 으로 이동한 값. \textbf{반응 과전압}은
구동 전위가 이 국소 평형에서 벗어난 정도다:
\begin{equation}
\boxed{\;\eta_j\equiv s\big[V_\drive-V_{\eq,j}(\xi_j)\big]
=s(V_\drive-U_j)-w_j\ln\frac{\xi_j}{1-\xi_j}\;.}
\label{eq:nch2_eta}
\end{equation}
\textbf{$\mathcal A_j^{\mathrm{net}}$ 과의 일치.} 기본형 $w_j=RT/F$ 에서 식~\eqref{eq:nch2_eta} 에 $F$ 를 곱하면
$F\eta_j=sF(V_\drive-U_j)-RT\ln[\xi_j/(1-\xi_j)]=\mathcal A_j-RT\ln[\xi_j/(1-\xi_j)]=\mathcal A_j^{\mathrm{net}}$ —
반응 과전압은 식~\eqref{eq:nch2_Anet} 의 net 친화도를 전위로 옮긴 \emph{같은 양}이며, 평형($\xi_j=\xi_{\eq,j}$)에서
$\eta_j=0$ 이다.

\textbf{계면 전류로의 변환.} 반응을 \emph{전하}의 흐름으로 보면, 평형($\eta_j=0$)에서 정$\cdot$역 반쪽 전류가 같은
크기 $i_{0,j}$(교환전류밀도)로 흐른다. 과전압이 걸리면 그것이 두 반쪽 장벽을 \emph{비대칭}으로 옮긴다 — 전이상태가
반응 좌표상 forward 쪽으로 $\alpha_j$ 만큼 위치하므로(식~(1.28) 의 장벽 분할과 같은 기제, $\alpha_j=\chi$), forward
장벽은 $\alpha_jF\eta_j$ 만큼 낮아지고 backward 장벽은 $(1-\alpha_j)F\eta_j$ 만큼 높아진다. 두 반쪽 전류의 차가
net 반응 전류밀도다 \cite{bardfaulkner,newman2004}:
\begin{equation}
\boxed{\;i_j=i_{0,j}\Big[\exp\!\Big(\frac{\alpha_jF\eta_j}{RT}\Big)-\exp\!\Big(-\frac{(1-\alpha_j)F\eta_j}{RT}\Big)\Big]\;.}
\label{eq:nch2_bv}
\end{equation}
두 전달계수의 합 $\alpha_j+(1-\alpha_j)=1$ 은 자유 선택이 아니라 정$\cdot$역이 \emph{같은 전이상태}를 공유함의
귀결이다(식~(1.28)$\to$(1.29)에서 합$=1$ 이 detailed balance 를 강제한 것과 같다).
\begin{verifybox}
\textbf{평형$\cdot$부호$\cdot$차원.} $\eta_j=0$ 이면 두 지수가 $1$ 이라 $i_j=0$(net 전류 소멸). $\eta_j>0$(forward
구동)이면 첫 지수 $>1$, 둘째 $<1$ 라 $i_j>0$(방전 방향). 지수 $F\eta_j/RT$ 는 무차원($\mathrm{(C/mol)\cdot V}
=\mathrm{J/mol}=RT$), $i_j$ 는 $i_{0,j}$ 와 같은 $\mathrm{A/m^2}$.
\end{verifybox}
\textbf{flux 와의 정합.} 전이 전류는 정의상 $I_j=Q_j\,\dd\xi_j/\dd t=Q_j(\mathcal J_j^+-\mathcal J_j^-)$ 이고 그
밀도가 $i_j=I_j/A_\eff$ 다(\S\ref{sec:nch2_current}). 식~\eqref{eq:nch2_bv} 의 Butler--Volmer 형은 이 밀도를
과전압으로 적은 것으로, 교환전류 $i_{0,j}$ 를 \emph{국소 상태} $\xi_j$ 에서 평가하면 단방향 flux 의 전하 표현과
정확히 같고, 평형 부근에서는 \S\ref{sec:nch2_i0} 의 $i_{0,j}(\xi_{\eq,j})$ 로 근사한다 — 어느 쪽이든 새 동역학이
아니라 같은 정$\cdot$역 속도론의 계면 표현이다.

\section{소신호 한계 — 전하전달 저항}\label{sec:nch2_rct}
% ====================================================================
작은 과전압 $|F\eta_j|\ll RT$ 에서 식~\eqref{eq:nch2_bv} 의 두 지수를 1차로 전개한다($e^x\simeq1+x$):
\begin{equation}
i_j\simeq i_{0,j}\Big[\big(1+\tfrac{\alpha_jF\eta_j}{RT}\big)-\big(1-\tfrac{(1-\alpha_j)F\eta_j}{RT}\big)\Big]
=i_{0,j}\,\frac{[\alpha_j+(1-\alpha_j)]F\eta_j}{RT}=i_{0,j}\,\frac{F\eta_j}{RT}.
\label{eq:nch2_lin}
\end{equation}
상수 $1$ 이 상쇄되고 전달계수 합이 $1$ 이라, 소신호 전류는 과전압에 \emph{선형}이며 $\alpha_j$ 에 무관하다(detailed
balance 의 flux 비가 분할 위치에 무관한 것과 같은 상쇄). 그 비례의 역수가 전하전달 저항이다:
\begin{equation}
\boxed{\;R_{\ct,j}=\frac{\partial\eta_j}{\partial i_j}\Big|_{\eta_j\to0}=\frac{RT}{F\,i_{0,j}}\;.}
\label{eq:nch2_rct}
\end{equation}
\begin{verifybox}
\textbf{차원.} $RT/(F\,i_{0,j})=\mathrm{(J/mol)}/[\mathrm{(C/mol)(A/m^2)}]=\mathrm{(J/C)/(A/m^2)}
=\mathrm V/(\mathrm{A/m^2})=\Omega\,\mathrm{m^2}$(면적 비저항). 전류로 환산하면 $R_{\ct,j}/A_\eff$ [$\Omega$].
\end{verifybox}
$R_{\ct,j}$ 는 \emph{전하전달 한 성분}의 저항이다 — 측정 전위축을 미는 분극 계수 $R_n$(식~(1.3),
$V_\app-V_n=s_I|I|R_n$)에는 ohmic$\cdot$농도분극$\cdot$film 이 함께 섞이므로, $R_{\ct,j}$ 는 그 일부이지 $R_n$
자체가 아니다(둘의 분리는 다채널 측정이 필요).

\section{교환전류밀도와 온도$\cdot$조성 의존}\label{sec:nch2_i0}
% ====================================================================
교환전류밀도 $i_{0,j}$ 는 평형($\eta_j=0$)에서의 정$=$역 반쪽 전류 크기다. 평형에서 두 단방향 flux 가 같으므로
($\mathcal J_j^+=\mathcal J_j^-=\mathcal J_{0,j}$), forward flux $\mathcal J_{0,j}=r_j^+(1-\xi_{\eq,j})$ 에 활성화
속도 $r_j^+=k_0\,e^{-(\Delta G_{a,j}-\chi\mathcal A_j)/RT}$(식~(1.28))를 넣는다. 평형에서는 detailed balance
(식~(1.29))로 $e^{\mathcal A_j/RT}=\xi_{\eq,j}/(1-\xi_{\eq,j})$ 이므로 $e^{\chi\mathcal A_j/RT}=[\xi_{\eq,j}
/(1-\xi_{\eq,j})]^{\chi}$ 이고, 따라서 $\mathcal J_{0,j}\propto k_0e^{-\Delta G_{a,j}/RT}\,\xi_{\eq,j}^{\,\chi}
(1-\xi_{\eq,j})^{1-\chi}$ 다. 전류로 환산하면($\alpha_j=\chi$)
\begin{equation}
i_{0,j}(q,T)=i_{0,j}^{\mathrm{ref}}\;g_j(\xi_{\eq,j})\;\exp\!\Big[-\frac{E_{a,j}}{R}\Big(\frac1T-\frac1{T_{\mathrm{ref}}}\Big)\Big],
\qquad g_j(\xi)=\xi^{\,\alpha_j}(1-\xi)^{\,1-\alpha_j},
\label{eq:nch2_i0}
\end{equation}
형이다($E_{a,j}$=교환전류의 유효 활성화 에너지, $g_j$=조성 활성도 인자). \textbf{조성 의존의 함의
($0<\alpha_j<1$).} $g_j(\xi)$ 는 $\dd\ln g_j/\dd\xi=\alpha_j/\xi-(1-\alpha_j)/(1-\xi)=0$ 에서, 곧 $\xi=\alpha_j$
에서 극대이고(대칭 $\alpha_j=\tfrac12$ 면 전이 중간 $\xi=\tfrac12$), 양 끝($\xi\to0,1$)에서 $0$ 으로 떨어진다. 따라서
교환전류는 전이 중간에서 가장 크고 \emph{양극단에서 작아}, 식~\eqref{eq:nch2_rct} 로 전하전달 저항 $R_{\ct,j}$ 가
\emph{전이 끝에서 커진다} — stage 경계 부근에서 전하전달이 느려져 분극이 커지는 거동과 정합한다.
\begin{boundbox}
\textbf{온도 의존.} 식~\eqref{eq:nch2_i0} 의 Arrhenius 인자로 저온에서 $i_{0,j}$ 가 작아져 $R_{\ct,j}$ 가
커진다($E_{a,j}>0$). 이는 활성화 속도 $r_j^\pm$ 가 저온서 급감하는 것과 같은 근원이며, 같은 전이의 활성화
배리어가 \emph{속도}(완화)와 \emph{전류}(교환)에 일관되게 나타남을 뜻한다.
\end{boundbox}

\section{비평형 정상상태와 평형 이탈 조건}\label{sec:nch2_ss}
% ====================================================================
정상 진행률은 진행이 멈추는 점, $\dd\xi_j/\dd t=0$ 의 해다. 식~\eqref{eq:nch2_Jpm} 에서 \emph{detailed balance 를
가정하지 않고}
\begin{equation}
r_j^+(1-\xi_{\sstat,j})=r_j^-\xi_{\sstat,j}\ \Longrightarrow\
\xi_{\sstat,j}=\frac{r_j^+}{r_j^++r_j^-}=\frac{1}{1+r_j^-/r_j^+}
\label{eq:nch2_xiss}
\end{equation}
을 얻는다. 여기 정$\cdot$역 비 $r_j^-/r_j^+$ 를 무엇으로 넣느냐가 결과를 가른다.

\textbf{두 경우.} (i) \emph{detailed balance 가 성립}하면 $r_j^+/r_j^-=e^{\mathcal A_j/RT}$(식~(1.29))이라
\begin{equation}
\xi_{\sstat,j}=\frac{1}{1+e^{-\mathcal A_j/RT}}=\xi_{\eq,j},
\label{eq:nch2_ss_eq}
\end{equation}
즉 정상 진행률이 \emph{평형값과 일치}한다(기본형 $V_\drive=V_n$ 에서 식~(1.11)과 동형) — 방향에 무관,
히스테리시스 없음. (ii) detailed balance 가 깨져 비가 $r_j^+/r_j^-=e^{\mathcal A_j/RT}\,C_j$ ($C_j\ne1$)가 되는
경우다. 이 $C_j$ 는 정$\cdot$역이 \emph{같은 전이상태}를 공유하지 않거나(비대칭 활성도) 분기에 따라 중심 전위가
이동할 때 생기며, $RT\ln C_j$ 는 중심 전위의 이동 $U_j\to U_j-(RT/sF)\ln C_j$ 로 흡수된다 — 곧 식~\eqref{eq:nch2_Anet}
의 net 친화도가 $\mathcal A_j^{\mathrm{net}}\to\mathcal A_j+RT\ln C_j-RT\ln[\xi_j/(1-\xi_j)]$ 로 함께 이동한다.
그러면 정상 진행률은
\begin{equation}
\xi_{\sstat,j}=\frac{1}{1+C_j^{-1}e^{-\mathcal A_j/RT}}\ \ne\ \xi_{\eq,j},
\label{eq:nch2_ss_neq}
\end{equation}
로 평형값에서 \emph{벗어난다}.

\begin{keybox}
\textbf{히스테리시스의 동역학적 기원.} 평형 진행률 $\xi_{\eq,j}$ 는 평형 전위가 정하는 열역학량이라 충$\cdot$방전에
무관하다. 그러나 정상 진행률 $\xi_{\sstat,j}$ 는 정$\cdot$역 비를 통해 \emph{동역학}에 의존하므로, detailed balance
가 깨지는 인자 $C_j$ 가 충전과 방전에서 다르면 두 방향의 정상 진행률이 갈린다 — 이것이 같은 평형 곡선 위에서도
충$\cdot$방전 궤적이 갈라지는 \emph{kinetic 히스테리시스}의 뿌리다. 어떤 물리가 $C_j$ 를 $1$ 에서 벗어나게
하는지는 강구동$\cdot$분기 거동에서 구체화된다.
\end{keybox}

\section{전이 전류와 전류 분배}\label{sec:nch2_current}
% ====================================================================
앞서 본 계면 전류와 전이 진행을 외부 전류와 잇는다. 전하 보존식 $Q_\cell q=Q_\bg(V_n,T)+\sum_j Q_j\xi_j$
(식~(1.1))의 양변을 시간 미분하면(연쇄율; $Q_\bg$ 는 $(V_n,T)$ 의 함수)
\begin{equation}
Q_\cell\frac{\dd q}{\dd t}=\frac{\partial Q_\bg}{\partial V_n}\frac{\dd V_n}{\dd t}
+\frac{\partial Q_\bg}{\partial T}\frac{\dd T}{\dd t}+\sum_{j=1}^{N_p} Q_j\frac{\dd\xi_j}{\dd t}.
\label{eq:nch2_balance_dt}
\end{equation}
등온 정전류 방전($\dd T/\dd t=0$, $Q_\cell\,\dd q/\dd t=|I|$)에서 좌변 $=|I|$ 이고, 우변을 배경 진행 전류와 전이
전류로 묶으면
\begin{equation}
\boxed{\;|I|=I_\bg+\sum_{j=1}^{N_p}I_j,\qquad I_\bg=\frac{\partial Q_\bg}{\partial V_n}\frac{\dd V_n}{\dd t},\qquad
I_j=Q_j\frac{\dd\xi_j}{\dd t}\;.}
\label{eq:nch2_current}
\end{equation}
\begin{verifybox}
\textbf{차원$\cdot$부호.} $I_j=Q_j[\mathrm C]\cdot\dd\xi_j/\dd t[\mathrm{s^{-1}}]=\mathrm A$;
$I_\bg=(\partial Q_\bg/\partial V_n)[\mathrm{C/V}]\cdot\dd V_n/\dd t[\mathrm{V/s}]=\mathrm A$. 방전 단일방향
($s=+1$)에서 모든 전이가 같은 방향이라 $\dd\xi_j/\dd t\ge0\Rightarrow I_j\ge0$; 배경 미분용량
$\partial Q_\bg/\partial V_n=C_\bg\ge\epsilon_Q>0$(식~(1.2))이라 $I_\bg$ 부호는 $\dd V_n/\dd t$ 가 정하며,
$I_\bg\ge0$ 인 영역에서 $\sum_j I_j\le|I|$ 다.
\end{verifybox}
이 $I_j=Q_j\,\dd\xi_j/\dd t$ 가 식~\eqref{eq:nch2_bv} 의 계면 전류와 $i_j=I_j/A_\eff$ 로 이어지며, 전이별
진행을 외부에서 흘린 전하에 귀속시킨다. 비등온에서는 $(\partial Q_\bg/\partial T)(\dd T/\dd t)$ 항이 더해지므로,
온도 변화에 따른 배경 용량 이동을 반응 전류로 오계상하지 않도록 progress 와 분리해야 한다.
